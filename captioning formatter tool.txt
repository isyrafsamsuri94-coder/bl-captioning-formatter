import re
from dataclasses import dataclass
from typing import List, Set, Optional, Tuple

import spacy

nlp = spacy.load("en_core_web_sm")

from spacy.matcher import PhraseMatcher, Matcher

# -----------------------------
# Config
# -----------------------------
@dataclass
class SubtitleCfg:
    max_len: int = 60          # max characters per line
    max_lines: int = 2         # subtitle blocks: 1–2 lines
    block_budget: int = 120    # approx 2 lines * 60; used for packing
    preserve_paragraph_breaks: bool = True


# -----------------------------
# Normalization (preserve punctuation and words)
# -----------------------------
def normalize_ws(text: str) -> str:
    """Normalize whitespace only; preserve punctuation and words."""
    return re.sub(r"[ \t]+", " ", text).strip()

def normalize_for_compare(text: str) -> str:
    """
    Normalization used ONLY for validation comparisons:
    - collapse whitespace and newlines to single spaces
    - keep punctuation unchanged
    """
    text = re.sub(r"\s+", " ", text).strip()
    return text

# --- 1) Curated phrase list (add as many as you want) ---
NEVER_BREAK_PHRASES = [
    "as well as",
    "in order to",
    "due to",
    "because of",
    "as a result",
    "in terms of",
    "according to",
    "rather than",
    "even though",
    "with respect to",
    "in relation to",
    "based on",
    "at least",
    "more than",
    "less than",
    "no more than",
    "as soon as possible",
    "at the end of the day",
    "in the meantime",
    "for the most part",
    "more or less",
    "goes through"
]

TITLES = {"mr", "mrs", "ms", "dr", "prof", "professor", "sir", "madam"}

UNITS = {
    "percent", "%", "km", "m", "cm", "mm", "kg", "g", "mg", "hz",
    "seconds", "second", "mins", "min", "minutes", "minute", "hours", "hour",
    "dollars", "dollar", "usd", "sgd", "eur", "gbp"
}

MONTHS = {
    "january","february","march","april","may","june","july","august",
    "september","october","november","december"
}

def protected_spans(doc) -> list:
    """
    Returns a list of spaCy Span objects that should not be split internally.
    Built from:
      - Exact phrases (PhraseMatcher)
      - Pattern rules (Matcher)
      - NER entities
      - PROPN sequences (simple)
    """
    spans = []

    # A) Exact phrase list
    phrase_matcher = PhraseMatcher(nlp.vocab, attr="LOWER")
    phrase_matcher.add("NEVER_BREAK_PHRASES", [nlp.make_doc(p) for p in NEVER_BREAK_PHRASES])
    for _, start, end in phrase_matcher(doc):
        spans.append(doc[start:end])

    # B) Pattern rules
    matcher = Matcher(nlp.vocab)

    # Title + Proper Noun(s): "Dr John Smith", "Professor Tan"
    matcher.add("TITLE_NAME", [[
        {"LOWER": {"IN": list(TITLES)}},
        {"POS": "PROPN", "OP": "+"}
    ]])

    # Number + Unit: "10 percent", "5 km", "$ 10 million" (simple coverage)
    matcher.add("NUM_UNIT", [[
        {"LIKE_NUM": True},
        {"LOWER": {"IN": list(UNITS)}, "OP": "+"}
    ]])

    # Date: "18 March 2026" / "March 18, 2026" (basic)
    matcher.add("DATE_DMY", [[
        {"LIKE_NUM": True},
        {"LOWER": {"IN": list(MONTHS)}},
        {"LIKE_NUM": True, "OP": "?"}
    ]])
    matcher.add("DATE_MDY", [[
        {"LOWER": {"IN": list(MONTHS)}},
        {"LIKE_NUM": True},
        {"IS_PUNCT": True, "OP": "?"},
        {"LIKE_NUM": True, "OP": "?"}
    ]])

    matches = matcher(doc)
    for _, start, end in matches:
        spans.append(doc[start:end])

    # C) Named entities (multiword)
    # Keep PERSON/ORG/GPE/LOC/PRODUCT etc. together
    for ent in doc.ents:
        if ent.label_ in {"PERSON","ORG","GPE","LOC","PRODUCT","EVENT","WORK_OF_ART","LAW"}:
            if len(ent) >= 2:
                spans.append(ent)

    # D) PROPN sequences (extra safety)
    # e.g., "National University of Singapore"
    i = 0
    while i < len(doc):
        if doc[i].pos_ == "PROPN":
            j = i + 1
            while j < len(doc) and (doc[j].pos_ == "PROPN" or doc[j].lower_ in {"of","and","the"}):
                j += 1
            if j - i >= 2:
                spans.append(doc[i:j])
            i = j
        else:
            i += 1

    # Optional: merge overlapping spans (keeps forbiddance robust)
    spans = spacy.util.filter_spans(spans)
    return spans

def forbid_breaks_inside_spans(spans) -> Set[int]:
    """
    For each span, forbid a break after every token inside it except the last token.
    """
    forbid = set()
    for sp in spans:
        for tok in sp[:-1]:
            forbid.add(tok.i)
    return forbid

# -----------------------------
# Phrase protection rules via spaCy
# -----------------------------
def forbidden_break_token_indices(doc) -> Set[int]:
    """
    Returns token indices i where a break AFTER token i is forbidden.
    Enforces:
      - Prepositional phrases
      - Compound nouns
      - Verb phrases (aux/neg/prt/to-infinitive cohesion)
    """
    forbid: Set[int] = set()

    # A) Prepositional phrases: forbid breaks inside the preposition subtree
    for tok in doc:
        if tok.pos_ == "ADP" or tok.dep_ == "prep":
            subtree = sorted(list(tok.subtree), key=lambda t: t.i)
            for a, b in zip(subtree, subtree[1:]):
                forbid.add(a.i)

    # B) Compound nouns
    for tok in doc:
        if tok.dep_ == "compound" and tok.head.pos_ in ("NOUN", "PROPN"):
            forbid.add(tok.i)

    # C) Verb phrases
    for tok in doc:
        if tok.dep_ in ("aux", "auxpass") and tok.head.pos_ == "VERB":
            forbid.add(tok.i)
        if tok.dep_ == "neg" and tok.head.pos_ == "VERB":
            forbid.add(tok.i)
        if tok.dep_ == "prt" and tok.head.pos_ == "VERB":
            forbid.add(tok.head.i if tok.head.i < tok.i else tok.i)
        if tok.lower_ == "to" and tok.pos_ == "PART":
            forbid.add(tok.i)

    # --- Possessive noun protection ---
    for tok in doc:
        # Case 1: possessive modifier relationship (student -> answer)
        # Prevent: "student's | answer"
        if tok.dep_ == "poss" and tok.head.pos_ in ("NOUN", "PROPN"):
            forbid.add(tok.i)

            # Also try to protect "'s" if it appears immediately after the possessor
            # Prevent: "student | 's"
            if tok.i + 1 < len(doc) and doc[tok.i + 1].tag_ == "POS":
                forbid.add(tok.i)  # break after possessor forbidden (already)
                forbid.add(doc[tok.i + 1].i)  # also avoid breaking after "'s" (before head)

        # Case 2: explicit clitic token "'s" (tag POS)
        # Prevent: "'s | answer"
        if tok.tag_ == "POS":
            forbid.add(tok.i)
    
    # --- Adverbial phrase protection (advmod subtrees) ---
    for tok in doc:
        if tok.dep_ == "advmod":
            subtree = sorted(list(tok.subtree), key=lambda t: t.i)
            if len(subtree) >= 2:
                # forbid breaks inside the adverbial phrase subtree
                for a, b in zip(subtree, subtree[1:]):
                    forbid.add(a.i)


    # Case 3: possessive determiners: my/your/his/her/its/our/their + NOUN
    # Prevent: "their | policy"
    poss_det = {"my","your","his","her","its","our","their","whose"}
    for i, tok in enumerate(doc[:-1]):
        if tok.lower_ in poss_det and doc[i+1].pos_ in ("NOUN","PROPN","ADJ"):
            forbid.add(tok.i)

    
    # --- Never end a line with 'or' (force 'or' to start next line) ---
    for tok in doc:
        if tok.lower_ == "or" and tok.whitespace_:
            # forbid break AFTER "or"
            forbid.add(tok.i)
            print(f"Forbidding break after 'or' at token index {tok.i}")

    # --- NEW: Protected spans (phrase lists + patterns + NER) ---
    spans = protected_spans(doc)
    forbid |= forbid_breaks_inside_spans(spans)

    return forbid

def is_break_after_function_word(line1: str) -> bool:
    """Extra guard: avoid breaks after dangling function words."""
    return bool(re.search(r"\b(of|to|in|on|at|for|as|by|with|and|or|but)\s*$", line1.lower()))

def punctuation_bonus(line1: str) -> int:
    """Prefer breaking after punctuation (readability)."""
    if re.search(r"[.!?]\s*$", line1): return 12
    if re.search(r"[;:]\s*$", line1):  return 8
    if re.search(r",\s*$", line1):     return 4
    return 0

def rebalance_dangling_function_word(left: str, right: str) -> Tuple[str, str]:
    """
    If left ends with a function word and right has content,
    move the last word from left to the start of right.
    """
    left = normalize_ws(left)
    right = normalize_ws(right)
    if not left or not right:
        return left, right

    m = re.search(r"\b(of|to|in|on|at|for|as|by|with|and|or|but)\s*$", left.lower())
    if not m:
        return left, right

    parts = left.split()
    if len(parts) <= 1:
        return left, right

    moved = parts[-1]
    new_left = " ".join(parts[:-1])
    new_right = f"{moved} {right}"
    return new_left, new_right


# -----------------------------
# Best 1–2 line formatting for a single block
# -----------------------------
def best_two_line_split(block_text: str, cfg: SubtitleCfg) -> Optional[str]:
    """
    Returns:
      - single line if <= max_len
      - "line1\nline2" if can split safely within 2 lines
      - None if cannot fit (rare; triggers fallback chunking)
    """
    text = normalize_ws(block_text)

    if len(text) <= cfg.max_len:
        return text

    doc = nlp(text)
    forbid_tok = forbidden_break_token_indices(doc)

    best = None
    best_pen = float("inf")

    for tok in doc[:-1]:
        if not tok.whitespace_:
            continue
        if tok.i in forbid_tok:
            continue

        b = tok.idx + len(tok.text)
        line1 = text[:b].rstrip()
        line2 = text[b:].lstrip()

        if len(line1) <= cfg.max_len and len(line2) <= cfg.max_len:
            pen = abs(len(line1) - len(line2))
            if is_break_after_function_word(line1):
                pen += 10
            pen -= punctuation_bonus(line1)

            if pen < best_pen:
                best_pen = pen
                best = f"{line1}\n{line2}"

    return best


# -----------------------------
# Transcript segmentation + packing
# -----------------------------
def split_into_paragraphs(text: str, preserve: bool) -> List[str]:
    text = text.strip()
    if not preserve:
        return [normalize_ws(text)]
    # preserve paragraph breaks as hard boundaries
    paras = [p.strip() for p in re.split(r"\n\s*\n", text) if p.strip()]
    return paras

def sentence_or_clause_units(paragraph: str) -> List[str]:
    """
    Create units to pack into blocks.
    - Use spaCy sentence segmentation
    - If a 'sentence' is huge or punctuation-poor, split by clause punctuation
    """
    para = normalize_ws(paragraph)
    if not para:
        return []

    doc = nlp(para)
    sents = [s.text.strip() for s in doc.sents if s.text.strip()]
    if not sents:
        sents = [para]

    units: List[str] = []
    for s in sents:
        # If very long, split further by punctuation boundaries
        if len(s) > 180:
            parts = re.split(r"(?<=[,;:.!?])\s+", s)
            parts = [p.strip() for p in parts if p.strip()]
            units.extend(parts if parts else [s])
        else:
            units.append(s)

    return units

def pack_units_into_blocks(units: List[str], cfg: SubtitleCfg) -> List[str]:
    """
    Pack consecutive units into rough blocks within cfg.block_budget (≈120 chars).
    """
    blocks: List[str] = []
    buf = ""

    for u in units:
        if not buf:
            buf = u
            continue

        cand = f"{buf} {u}"
        if len(cand) <= cfg.block_budget:
            buf = cand
        else:
            left, right = rebalance_dangling_function_word(buf, u)
            blocks.append(left)
            buf = right

    if buf:
        blocks.append(buf)

    return blocks


# -----------------------------
# Last-resort fallback when something won't fit
# -----------------------------
def hard_wrap_to_blocks(text: str, cfg: SubtitleCfg) -> List[str]:
    """
    Last resort: length-based splitting into multi blocks.
    Only breaks at whitespace.
    """
    text = normalize_ws(text)
    if not text:
        return []

    words = text.split(" ")
    blocks: List[str] = []
    buf = ""

    def flush(chunk: str):
        chunk = chunk.strip()
        if not chunk:
            return
        formatted = best_two_line_split(chunk, cfg)
        if formatted is not None:
            blocks.append(formatted)
        else:
            # blunt cut if needed
            cut = chunk[:cfg.max_len].rstrip()
            rest = chunk[cfg.max_len:].lstrip()
            blocks.append(cut + "\n" + rest)

    for w in words:
        if not buf:
            buf = w
        else:
            cand = buf + " " + w
            if len(cand) <= cfg.block_budget:
                buf = cand
            else:
                left, right = rebalance_dangling_function_word(buf, w)
                flush(left)
                buf = right

    if buf:
        flush(buf)

    return blocks

def remove_trailing_commas_from_blocks(blocks):
    """
    blocks: list[str]
      - each block is either "one line" or "line1\nline2"
      - paragraph separators can be "" (leave untouched)
    """
    out = []
    for block in blocks:
        if block == "":
            out.append(block)
            continue

        lines = block.split("\n")
        lines = [re.sub(r",\s*$", "", line) for line in lines]
        out.append("\n".join(lines))
    return out

# -----------------------------
# Rules-first formatter for whole transcripts
# -----------------------------
def format_transcript_rules(text: str, cfg: SubtitleCfg) -> List[str]:
    """
    Main rules-first entry point.
    Returns a list of blocks, each block is 1 line or 2 lines separated by '\n'.
    """
    blocks_out: List[str] = []
    paragraphs = split_into_paragraphs(text, cfg.preserve_paragraph_breaks)

    for para in paragraphs:
        units = sentence_or_clause_units(para)
        rough_blocks = pack_units_into_blocks(units, cfg)

        for blk in rough_blocks:
            formatted = best_two_line_split(blk, cfg)
            if formatted is not None:
                blocks_out.append(formatted)
            else:
                # If can't fit, try further clause splits then hard wrap as last resort
                parts = re.split(r"(?<=[,;:.!?])\s+", blk)
                parts = [p.strip() for p in parts if p.strip()]
                if parts and len(parts) > 1:
                    for p in parts:
                        f2 = best_two_line_split(p, cfg)
                        if f2 is not None:
                            blocks_out.append(f2)
                        else:
                            blocks_out.extend(hard_wrap_to_blocks(p, cfg))
                else:
                    blocks_out.extend(hard_wrap_to_blocks(blk, cfg))

        # Insert a blank block to preserve paragraph boundary visually (optional)
        if cfg.preserve_paragraph_breaks:
            blocks_out.append("")  # paragraph separator

    # Remove trailing separators
    while blocks_out and blocks_out[-1] == "":
        blocks_out.pop()

    return blocks_out

def blocks_to_plain_text(blocks: List[str]) -> str:
    """
    Join blocks into plain text with blank line between blocks.
    Empty string blocks create extra blank lines (paragraph breaks).
    """
    out_lines = []
    for b in blocks:
        if b == "":
            out_lines.append("")  # paragraph boundary
            continue
        out_lines.append(b)
        out_lines.append("")      # blank line after each subtitle block
    return "\n".join(out_lines).rstrip()


# -----------------------------
# Optional AI step + strict validation
# -----------------------------
def allowed_break_positions(text: str) -> Set[int]:
    """
    Allowed break positions are token end-char positions where:
    - there is whitespace after the token
    - boundary is not forbidden by phrase rules
    """
    text = normalize_ws(text)
    doc = nlp(text)
    forbid = forbidden_break_token_indices(doc)

    allowed: Set[int] = set()
    for tok in doc[:-1]:
        if tok.whitespace_ and tok.i not in forbid:
            allowed.add(tok.idx + len(tok.text))
    return allowed

def validate_ai_output(original: str, ai_text: str, cfg: SubtitleCfg) -> Tuple[bool, List[str]]:
    reasons = []
    orig = normalize_for_compare(original)
    ai_cmp = normalize_for_compare(ai_text.replace("\n", ""))
    if ai_cmp != orig:
        reasons.append("AI changed characters (beyond inserting line breaks).")
        return False, reasons

    # Check per-line max length
    for line in ai_text.split("\n"):
        if line.strip() and len(line) > cfg.max_len:
            reasons.append(f"Line exceeds {cfg.max_len} chars: {line!r}")

    # Check 2 lines per block (blocks separated by blank lines)
    blocks = [b for b in ai_text.split("\n\n") if b.strip()]
    for b in blocks:
        lines = [ln for ln in b.split("\n") if ln.strip()]
        if len(lines) > cfg.max_lines:
            reasons.append(f"Block has > {cfg.max_lines} lines: {b!r}")

    # Check breaks are allowed positions
    allowed = allowed_break_positions(original)
    # Map newline positions to original char index (counting non-newline chars)
    idx = 0
    for ch in ai_text:
        if ch == "\n":
            if idx not in allowed:
                reasons.append(f"Break at char {idx} not allowed (likely splits protected phrase).")
        else:
            idx += 1

    return (len(reasons) == 0), reasons

def looks_awkward(block: str) -> bool:
    """Heuristic trigger for AI refinement."""
    lines = [ln for ln in block.split("\n") if ln.strip()]
    if len(lines) == 2:
        return abs(len(lines[0]) - len(lines[1])) > 25
    return False

def llm_refine_stub(original_text: str, cfg: SubtitleCfg) -> str:
    """
    Replace with your LLM call (Azure OpenAI / etc.).
    Must return plain text with:
    - '\n' separating lines within a block
    - blank line between blocks
    """
    # Stub returns rules output
    blocks = format_transcript_rules(original_text, cfg)
    return blocks_to_plain_text(blocks)

def format_transcript_hybrid(text: str, cfg: SubtitleCfg, use_ai: bool = True) -> str:
    """Rules-first, optionally AI-refine, always validate AI output, fallback to rules."""
    blocks = format_transcript_rules(text, cfg)
    blocks = remove_trailing_commas_from_blocks(blocks)
    rule_text = blocks_to_plain_text(blocks)

    if not use_ai:
        return rule_text

    # Trigger AI only if any block looks awkward
    if not any(looks_awkward(b) for b in blocks if b.strip()):
        return rule_text

    ai_text = llm_refine_stub(text, cfg)
    ok, reasons = validate_ai_output(text, ai_text, cfg)
    if ok:
        return ai_text

    # fallback safely
    return rule_text
